#include<stdio.h>
#include<string.h>
int main()
{
    char data[100],divisor[20];
    int datalen,divlen;
    int temp[120],i,j;
    printf("enter data");
    scanf("%s",data);
    printf("enter divisor");
    scanf("%s",divisor);
    datalen=strlen(data);
    divlen=strlen(divisor);
    for(i=0;i<datalen;i++)
        {
            temp[i]=data[i]-'0';
            
        }
    for(i=0;i<divlen-1;i++)
        {
            temp[datalen+i]=0;
        }
    int total=datalen+divlen-1;
    for(i=0;i<total-divlen;i++)
        {
            if(temp[i]==1)
            {
                for(j=0;j<divlen;j++)
                    {
                        temp[i+j]^=(divisor[j]-'0');
                        
                    }
            }
            
        }
    printf("\n crc bits:");
    for(i=datalen;i<total;i++)
        {
            printf("%d",temp[i]);
        }
    printf("transmitted data:");
    for(i=0;i<datalen;i++)
        {
            printf("%c",data[i]);
        }
    for(i=datalen;i<total;i++)
        {
            printf("%d",temp[i]);
        }
    printf("\n");
    return 0;
}
